c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    char name[50];
    int age;
    float height;
} Person;

typedef struct {
    Person *persons;
    int count;
} t_struct;

t_struct initialize_t_struct(int argc, char **argv) {
    t_struct result;
    result.count = argc - 1; // Number of persons is the number of arguments minus the program name
    result.persons = malloc(result.count * sizeof(Person));

    // Initialize each person from command-line arguments
    for (int i = 1; i < argc; i++) { // Start from 1 to skip program name
        sscanf(argv[i], "%s %d %f", result.persons[i - 1].name, &result.persons[i - 1].age, &result.persons[i - 1].height);
    }

    return result;
}

int main(int argc, char **argv) {
    if (argc < 2) {
        printf("Usage: %s name1 age1 height1 [name2 age2 height2 ...]\n", argv[0]);
        return 1;
    }

    t_struct persons = initialize_t_struct(argc, argv);

    // Printing the details of each person
    printf("Persons:\n");
    for (int i = 0; i < persons.count; i++) {
        printf("Name: %s, Age: %d, Height: %.2f\n", persons.persons[i].name, persons.persons[i].age, persons.persons[i].height);
    }

    // Don't forget to free the allocated memory
    free(persons.persons);

    return 0;
}

To create multiple threads, each containing different values, you can use pthreads library in C. Each thread can have its own set of arguments passed to it. Here's a basic example:

c
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

#define NUM_THREADS 5

// Structure to hold the arguments for each thread
typedef struct {
    int thread_id;
    char *message;
    int value;
} ThreadArgs;

// Function that each thread will execute
void *thread_function(void *args) {
    ThreadArgs *thread_args = (ThreadArgs *) args;
    printf("Thread %d: %s, Value: %d\n", thread_args->thread_id, thread_args->message, thread_args->value);
    pthread_exit(NULL);
}

int main() {
    pthread_t threads[NUM_THREADS];
    ThreadArgs thread_args[NUM_THREADS];

    // Creating threads
    for (int i = 0; i < NUM_THREADS; i++) {
        thread_args[i].thread_id = i;
        thread_args[i].message = "Hello from thread";
        thread_args[i].value = i * 10;
        int rc = pthread_create(&threads[i], NULL, thread_function, (void *)&thread_args[i]);
        if (rc) {
            printf("Error: Unable to create thread, %d\n", rc);
            exit(-1);
        }
    }

    // Waiting for all threads to finish
    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    pthread_exit(NULL);
    return 0;
}

In this example:

  • We define a struct ThreadArgs to hold the arguments for each thread. In this case, it contains the thread ID, a message, and a value.
  • We have a function thread_function that each thread will execute. It takes a void * argument which is casted to ThreadArgs * to access thread-specific arguments.
  • In the main function, we create an array of pthread_t to hold thread IDs and an array of ThreadArgs to hold arguments for each thread.
  • We create multiple threads in a loop. Each thread is assigned a unique thread ID, a message, and a value. The pthread_create function is used to create each thread, passing the thread function and its arguments.
  • After creating all threads, we wait for them to finish executing using pthread_join.
  • Finally, we exit the program.

When you run this program, you'll see output from each thread showing its thread ID, message, and value.
